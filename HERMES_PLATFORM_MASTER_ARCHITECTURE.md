// Backend (Node.js):
async function sendMessage(userId, aiTeamId, message) {
  // 1. Get encrypted prompt from DB
  const team = await db.ai_teams.findById(aiTeamId);
  
  // 2. Decrypt prompt (server-side only!)
  const prompt = decrypt(team.prompt_encrypted, SECRET_KEY);

  const response = await claude.messages.create({
    model: 'claude-sonnet-4-20250514',

    system: prompt, // ← Client NEVER sees this!

  // 4. Return ONLY the response (not the prompt)
  return { reply: response.content[0].text };
}

wss.on('connection', (ws, req) => {

});
